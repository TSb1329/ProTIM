Public Sub Formularaufruf(Formular As String, Übergabetext As String, Gebunden As Boolean)
    Dim i As Integer
    
    For i = 1 To FormularEbenen
        If FormularEbene(i) = "" Then
            FormularEbene(i) = Formular
            Exit For
        End If
    Next i
    If Gebunden Then
        If Übergabetext <> "ohne" Then
            DoCmd.OpenForm Formular, , , Übergabetext, , acDialog
        Else
            DoCmd.OpenForm Formular, , , , , acDialog
        End If
    Else
        If Übergabetext <> "ohne" Then
            DoCmd.OpenForm Formular, , , Übergabetext
        Else
            DoCmd.OpenForm Formular
        End If
    End If
    For i = 1 To (FormularEbenen - 1)
        If FormularEbene(i + 1) = "" Then
            FormularEbene(i) = ""
            Exit For
        End If
    Next i
End Sub

Public Function Start()

    Call ShowWindow(hWndAccessApp, 2)

    Dim projektPfad As String
    Dim letzterOrdner As String
    Dim basisPfad As String

    projektPfad = Application.CurrentProject.Path

    If Right(projektPfad, 1) = "\" Then
        projektPfad = Left(projektPfad, Len(projektPfad) - 1)
    End If

    letzterOrdner = Mid(projektPfad, InStrRev(projektPfad, "\") + 1)
    basisPfad = Left(projektPfad, InStrRev(projektPfad, "\"))

    Select Case letzterOrdner
        Case "ProTIM - Test"
            Laufwerkpfad = projektPfad & "\Projekte\"
            LaufwerkpfadBEDB = projektPfad & "\"
            Entwicklung = True

        Case "ProTIM"
            Laufwerkpfad = projektPfad & "\Projekte\"
            LaufwerkpfadBEDB = projektPfad & "\"
            Entwicklung = False

        Case Else
            GoTo ErrNoPath
    End Select

    NeuVerknüpfen
    DoCmd.OpenForm "F_ProjektListe"

    Exit Function

ErrNoPath:
    Hinweis "Fehlerhafter Pfad!"
    DoCmd.Quit

End Function

Public Sub NeuVerknüpfen()
    Dim db As Database
    Dim strPfadZurExterneDB As String
    Dim arrTabellennamen As Variant
    Dim strTabellenname As Variant
    Dim strNeueVerknüpfung As String
    
    ' Setze den Pfad zur externen Datenbank
    strPfadZurExterneDB = LaufwerkpfadBEDB & "ProTIM_BE.accdb"
    
    ' Definiere die Liste der Tabellennamen
    arrTabellennamen = Array("Abfallarten", "AktiveUser", "Artikel", "ArtikelConfig", "ArtikelKategorie", "ArtikelListe", "Attachments", "Bearbeiter", "Begleitscheine", "Boxen", "BoxenJournal", "Buchungsarten", "Kunden", "KundenAnsprechperson", "LagerPlätze", "Lieferscheine", "Messarten", "Projekte", "ProjektVorgang", "ProjektVorgänge", "ProjektVorgangGelöscht", "Transportarten", "Transporteure")
    
    ' Öffne die aktuelle Datenbank
    Set db = CurrentDb
    
    ' Durchlaufe die Liste der Tabellennamen
    For Each strTabellenname In arrTabellennamen
        ' Erstelle die Verbindungsinformationen für die neue Verknüpfung
        strNeueVerknüpfung = ";DATABASE=" & strPfadZurExterneDB
        
        ' Lösche die bestehende Verknüpfung, falls vorhanden
        On Error Resume Next
        db.TableDefs.Delete strTabellenname
        On Error GoTo 0
        
        ' Erstelle die neue Verknüpfung
        db.TableDefs.Append db.CreateTableDef(strTabellenname, dbAttachSavePWD, strTabellenname, strNeueVerknüpfung)
    Next strTabellenname
    
    ' Aktualisiere die Beziehungen und Ansichten
    db.Relations.Refresh
    db.QueryDefs.Refresh
    db.TableDefs.Refresh
    
    ' Schließe die Datenbank
    db.Close
    
End Sub

Public Function maxWert(Wert As String, Tabelle As String) As Integer
    Dim db As DAO.Database
    Dim rs As DAO.Recordset

    Set db = CurrentDb()
    
    On Error GoTo Fehlerbehandlung
    Set rs = db.OpenRecordset("SELECT Max([" & Wert & "]) AS MaxWert FROM [" & Tabelle & "]")

    If Not rs.EOF Then
        maxWert = Nz(rs!maxWert, 0)
    Else
        maxWert = 0
    End If

    rs.Close
    Set rs = Nothing
    Set db = Nothing
    Exit Function

Fehlerbehandlung:
    MsgBox "Fehler in maxWert: " & Err.Description, vbCritical
    maxWert = -1
End Function

Public Sub DirListe(DateienImPfad As ListBox)
    Dim strOrdnerPfad As String
    Dim strDatei As String

    ' Definiere den Pfad zum Ordner, dessen Inhalt du anzeigen möchtest
    strOrdnerPfad = Laufwerkpfad & Forms!F_ProjektBearbeitung.ProjektName & "\"
    
    ' Überprüfe, ob der angegebene Ordner existiert
    If Dir(strOrdnerPfad, vbDirectory) <> "" Then
        ' Schleife durch alle Dateien im Ordner
        strDatei = Dir(strOrdnerPfad & "*.*")
        DateienImPfad.RowSource = ""
        Do While strDatei <> ""
            ' Füge den Dateinamen dem Listenfeld hinzu
            DateienImPfad.AddItem strDatei
            ' Gehe zur nächsten Datei im Ordner
            strDatei = Dir
        Loop
    Else
        Hinweis "Der angegebene Ordner existiert nicht!"
    End If
End Sub

Public Sub EintragInOutlookKalender(Betreff As String, Beginn As Date, Ende As Date, Ort As String, TerminTxt As String)
    
    Dim objOutlook As Object
    Dim objNamespace As Object
    Dim objFolder As Object
    Dim objAppointment As Object
    
    ' Outlook Objekt erstellen
    Set objOutlook = CreateObject("Outlook.Application")
    Set objNamespace = objOutlook.GetNamespace("MAPI")
    
    ' Spezifisches Outlook-Konto auswählen
    On Error Resume Next
    Set objFolder = objNamespace.Folders("logistics@i4next.com").Folders("Kalender")
    Set objFolder = objNamespace.Folders("i4next Logistik").Folders("Kalender")
    On Error GoTo 0
    If objFolder Is Nothing Then
        Hinweis "Logistik Kalender wurde nicht gefunden!"
        Exit Sub
    End If
    
    'bestehenden Termin mit dem Betreff suchen und falls vorhanden dann löschen!
    For Each objAppointment In objFolder.Items
        If objAppointment.Class = 26 And objAppointment.Subject = Betreff Then ' 26 steht für Termine (Appointment)
            ' Eintrag mit dem gesuchten Betreff gefunden, lösche ihn
            objAppointment.Delete
            Exit For ' Die Schleife verlassen, da der Eintrag bereits gelöscht wurde
        End If
    Next objAppointment
    
    ' Neuen Termin im ausgewählten Kalender erstellen
    Set objAppointment = objFolder.Items.Add(1) ' 1 steht für einen Termin (Appointment)
    With objAppointment
        .Subject = Betreff  ' Betreff des Termins
        .Start = Beginn     ' Startdatum und -zeit des Termins
        .End = Ende         ' Enddatum und -zeit des Termins
        .Location = Ort     ' Ort des Termins
        .Body = Trim(TerminTxt)   ' Text des Termins
        
        ' Termin speichern
        .Save
    End With
    
    ' Objekte freigeben
    Set objAppointment = Nothing
    Set objFolder = Nothing
    Set objNamespace = Nothing
    Set objOutlook = Nothing

End Sub


Public Function EintragAusOutlookKalenderLesen(Betreff As String) As Variant
    
    Dim Kalendereintrag(1 To 5) As Variant
    
    Dim objOutlook As Object
    Dim objNamespace As Object
    Dim objFolder As Object
    Dim objItem As Object
    Dim SucheBetreff As String
    
    ' Betreff, den du suchen möchtest
    SucheBetreff = Betreff
    
    ' Outlook Objekt erstellen
    Set objOutlook = CreateObject("Outlook.Application")
    Set objNamespace = objOutlook.GetNamespace("MAPI")
    
    ' Spezifisches Outlook-Konto und Kalenderordner auswählen (ersetze "Kontoname" durch den Namen deines Kontos)
    On Error Resume Next
    Set objFolder = objNamespace.Folders("logistics@i4next.com").Folders("Kalender")
    Set objFolder = objNamespace.Folders("i4next Logistik").Folders("Kalender")
    On Error GoTo 0
    If objFolder Is Nothing Then
        Hinweis "Logistik Kalender wurde nicht gefunden!"
        Exit Function
    End If
    
    ' Durchlaufe die Items-Auflistung des Kalenders und suche nach dem Betreff
    For Each objItem In objFolder.Items
        If objItem.Class = 26 And objItem.Subject = SucheBetreff Then ' 26 steht für Termine (Appointment)
            ' Betreff wurde gefunden
            Kalendereintrag(1) = objItem.Subject
            Kalendereintrag(2) = objItem.Start
            Kalendereintrag(3) = objItem.End
            Kalendereintrag(4) = objItem.Location
            Kalendereintrag(5) = objItem.Body
        End If
    Next objItem
    
    ' Objekte freigeben
    Set objItem = Nothing
    Set objFolder = Nothing
    Set objNamespace = Nothing
    Set objOutlook = Nothing

    EintragAusOutlookKalenderLesen = Kalendereintrag

End Function

Public Sub Copy_ArtikelListe(TempProjektNr As Integer)
    Dim strSQL As String
    Dim db As DAO.Database
    Dim rsQuelle As DAO.Recordset
    
    DoCmd.SetWarnings False
    Set db = CurrentDb
    Set rsQuelle = db.OpenRecordset("SELECT * FROM ArtikelListeTemp ORDER BY AbholNr", dbOpenSnapshot)
    Do While Not rsQuelle.EOF
        If Nz(rsQuelle!Menge, 0) > 0 Then
            strSQL = "INSERT INTO ArtikelListe (ProjektNr, ArtikelKatID, Menge, Bemerkung) " & _
                        "VALUES (" & TempProjektNr & ", " & rsQuelle!ArtikelKatID & ", " & _
                        rsQuelle!Menge & ", " & _
                        "'" & rsQuelle!Bemerkung & "')"
            db.Execute strSQL, dbFailOnError
        End If
        rsQuelle.MoveNext
    Loop

    rsQuelle.Close
    Set rsQuelle = Nothing
    Set db = Nothing
    DoCmd.SetWarnings True
End Sub


Public Sub Renew_ArtikelListe(TempProjektNr As Integer)
    'löscht alle Artikel zu einer Projektnummer und schreibt die Artikel aus der ArtikelListeTemp neu
    Dim strSQL As String
    Dim db As DAO.Database
    Dim rsQuelle As DAO.Recordset
    
    DoCmd.SetWarnings False
    strSQL = "DELETE * FROM ArtikelListe WHERE ProjektNr = " & TempProjektNr
    DoCmd.RunSQL (strSQL)
    Set db = CurrentDb
    Set rsQuelle = db.OpenRecordset("SELECT * FROM ArtikelListeTemp ORDER BY AbholNr", dbOpenSnapshot)
    Do While Not rsQuelle.EOF
        If Nz(rsQuelle!Menge, 0) > 0 Then
            strSQL = "INSERT INTO ArtikelListe (ProjektNr, ArtikelKatID, Menge, Bemerkung) " & _
                        "VALUES (" & TempProjektNr & ", " & rsQuelle!ArtikelKatID & ", " & _
                        rsQuelle!Menge & ", " & _
                        "'" & rsQuelle!Bemerkung & "')"
            db.Execute strSQL, dbFailOnError
        End If
        rsQuelle.MoveNext
    Loop

    rsQuelle.Close
    Set rsQuelle = Nothing
    Set db = Nothing
    DoCmd.SetWarnings True
End Sub
